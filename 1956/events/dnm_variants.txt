country_event = {
	id = shipvariants.1
	title = shipvariants.1.t
	desc = shipvariants.1.d
	picture = GFX_welcome_to_dnm
	trigger = {
		NOT = {
			has_country_flag = dnmstarted
		}
	}

	is_triggered_only = yes

	immediate = {
		set_country_flag = dnmstarted
	}

	option = {
	ai_chance = { factor = 100 }
		name = shipvariants.1.a
		custom_effect_tooltip = shipvariants.1.a.tt
		hidden_effect = {
			ROOT = { 
			set_country_flag = historic_ship_variants
			}
		}
	}
	option = {
		trigger = {
			is_ai = no
		}
		name = shipvariants.1.b
		custom_effect_tooltip = shipvariants.1.b.tt
		hidden_effect = {
			ROOT = { 
			set_country_flag = ship_design_companies
			}
		}
	}
	option = {
		trigger = {
			is_ai = no
		}
		name = shipvariants.1.c
		custom_effect_tooltip = shipvariants.1.c.tt
		hidden_effect = {
			ROOT = { 
			set_country_flag = ship_design_companies
			set_country_flag = no_ship_events
			}
		}
	}
}

#Buenos Aires Class Destroyer (basic_destroyer)
country_event = {
	id = shipvariants.2
	title = shipvariants.2.t
	desc = shipvariants.2.d
	picture = GFX_buenos_aires_arg
	trigger = {
		tag = ARG
		OR = {
			is_ai = yes
			has_country_flag = historic_ship_variants
		}
		NOT = {
			has_country_flag = no_ship_events
		}
		has_country_flag = dnmstarted
	}

	is_triggered_only = yes

	fire_only_once = yes

	option = { 
		name = shipvariants.2.a
		create_equipment_variant = {
			name = "Buenos Aires Class"
			type = destroyer_2
			upgrades = {
				destroyer_engine_upgrade = 1
				ship_barrels_upgrade = 1
				ship_calibre_upgrade = 1
				ship_gunquality_upgrade = 0
				ship_quality_upgrade = 1
				ship_ASW_upgrade = 0
				ship_anti_air_upgrade = 0
				ship_range_upgrade = 0
				ship_torpedo_upgrade = 3
			}
		}
	}
}
