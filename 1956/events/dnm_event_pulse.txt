#Hidden event to use days = X random = X instead of MTTH for opening variant events
#First event sets up the annual pulses, which manage the monthly pulses.  There's a separate event for the first year's monthly pulses, as it's not a full year.
country_event = {
	id = dnmeventp.1

	hidden = yes

	is_triggered_only = yes

	immediate = {
		if = { #Checks whether nation has advanced ship designs (triggered event changes designer idea if so)
			limit = {
				OR = {
					has_country_flag = ship_design_companies
					has_country_flag = late_war_designer
				}
				NOT = {
					has_country_flag = advanced_ship_design_companies
				}
				OR = {
					has_tech = advanced_destroyer
					has_tech = advanced_submarine
					has_tech = advanced_light_cruiser
					has_tech = advanced_heavy_cruiser
					has_tech = advanced_carrier
				}
			}
			ROOT = { country_event { id = tpov_navaldesign.2 days = 5 random = 5 } }
		}
		if = { #Checks whether minor has enough NIC to upgrade their ship designer idea (triggered event upgrades if so)
			limit = {
				has_country_flag = ship_design_companies
				NOT = {
					OR = {
						tag = ENG
						tag = FRA
						tag = GER
						tag = ITA
						tag = JAP
						tag = SOV
						tag = USA
					}
				}
			}
			ROOT = { country_event { id = tpov_navaldesign.3 days = 5 random = 5 } }
		}
		if = { #Checks whether navy outnumbered by an opponent
			limit = {
				ROOT = { has_navy_size = { size > 1 } }
			}
			ROOT = { country_event { id = dnmeventp.16 days = 5 random = 5 } }
		}
		if = { #Runs checks of whether navy short of various types of ships and sets flags appropriately
			limit = {
				ROOT = { num_of_naval_factories > 1 }
			}
			ROOT = { country_event { id = dnmeventp.17 days = 5 random = 5 } }
		}
		if = { #Check to see whether a country has ceased to exist or not (so drops out of the monthly pulse)
			limit = {
				has_country_flag = tpov_exists
				exists = no
			}
		ROOT = { clr_country_flag = tpov_exists	}			
		}
		#ROOT = { country_event { id = dnmeventp.2 } } #Test event to check monthly pulse is working - tag out for actual playing (!) but handy to leave in for testing.
		if = { #Event to stop AI nations having any naval XP - they get historic stats.
			limit = {
				is_ai = yes
				ROOT = { num_of_naval_factories > 0 }
			}
			ROOT = { navy_experience = -500 }
		}
	}

	option = {}
}

country_event = { #Checks whether a country that didn't exist does now, and needs to get in on the monthly pulse action.
	id = dnmeventp.12

	hidden = yes

	trigger = {
		NOT = { has_country_flag = tpov_exists }
		exists = yes
	}

	mean_time_to_happen = { days = 30 }

	immediate = {
		ROOT = {
			set_country_flag = tpov_exists
			country_event = { id = dnmeventp.13 days = 30 } #Goes through the cycle again in 30 days time
			country_event = { id = dnmeventp.1 days = 1 random = 5 } #The event that actually checks things/fires
		}
	}

	option = {}
}
