#Series XIV (basic_submarine)
country_event = {
	id = shipvard.1
	title = shipvard.1.t
	desc = shipvard.1.d
	picture = GFX_ss_series_xiv_sov
	trigger = {
		tag = SOV
		NOT = { has_tech = improved_submarine }
		NOT = {
			has_country_flag = no_ship_events
		}
		has_country_flag = dnmstarted
	}

	is_triggered_only = yes

	option = {
		trigger = {
			OR = {
				is_ai = yes
				has_country_flag = historic_ship_variants
			}			
		}
		name = shipvard.1.a
		create_equipment_variant = {
			name = "Series XIV K Class"
			type = submarine_2
			parent_version = 0
			upgrades = {
				sub_engine_upgrade = 11
				ship_quality_upgrade = 10
				ship_anti_air_upgrade = 8
				ship_range_upgrade = 13
				sub_stealth_upgrade = 9
				sub_torpedo_upgrade = 14
			}
		}
	}

	option = {
		trigger = {
			is_ai = no
			has_country_flag = ship_design_companies
		}
		name = shipvard.1.b
		custom_effect_tooltip = shipvard.1.b.tt
	}
}
