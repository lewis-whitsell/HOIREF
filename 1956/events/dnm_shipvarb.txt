#Le Hardi Class (basic_destroyer)
country_event = {
	id = shipvarb.1
	title = shipvarb.1.t
	desc = shipvarb.1.d
	picture = GFX_dd_le_hardi_fra
	trigger = {
		tag = FRA
		NOT = { has_tech = improved_destroyer }
		#has_country_flag = marciliodiasdd
		NOT = {
			has_country_flag = no_ship_events
		}
		has_country_flag = dnmstarted
	}

	is_triggered_only = yes

	fire_only_once = yes

	option = {
		trigger = {
			OR = {
				is_ai = yes
				has_country_flag = historic_ship_variants
			}			
		}
		name = shipvarb.1.a
		create_equipment_variant = {
			name = "Le Hardi Class"
			type = destroyer_2
			parent_version = 0
			upgrades = {
				destroyer_engine_upgrade = 13
				ship_barrels_upgrade = 15
				ship_calibre_upgrade = 7
				ship_gunquality_upgrade = 2
				ship_quality_upgrade = 7
				ship_ASW_upgrade = 0
				ship_anti_air_upgrade = 0
				ship_range_upgrade = 0
				ship_torpedo_upgrade = 2
			}
		}
		set_country_flag = lehardidd
	}

	option = {
		trigger = {
			is_ai = no
			has_country_flag = ship_design_companies
		}
		name = shipvarb.1.b
		custom_effect_tooltip = shipvarb.1.b.tt
		set_country_flag = lehardidd
	}
}
