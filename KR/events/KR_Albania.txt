### Fate of Albania

country_event = {
	id = albania.2
	title = albania.2.t
	desc = albania.2.d

	fire_only_once = yes

	trigger = {
		tag = ALB
		is_subject_of = OTT
		OTT = {
			has_war = yes
			enemies_strength_ratio > 0.3
		}
		NOT = {
			has_war_with = SER
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	option = { # declare independence
		name = albania.2.a

		ai_chance = {
			factor = 99
		}
		OTT = {
			end_puppet = ALB
		}
		load_oob = ALB_1
		every_country = {
			limit = {
				has_war_with = ALB
			}
			white_peace_with_ROOT = yes
		}
		# TODO - Zog flag
		set_politics = {
			ruling_party = paternal_autocrat
			elections_allowed = no
		}
		retire_country_leader = yes
		create_country_leader = {
			name = "Zog I"
			picture = "Portrait_Albania_Zog_I.tga"
			expire = "1965.1.1"
			ideology = paternal_autocrat_subtype
			traits = {}
		}
		OTT = {
			country_event = {
				id = albania.3
				days = 0
			}
		} #ottoman response
	}

	option = { # remain a puppet
		name = albania.2.b

		ai_chance = {
			factor = 1
		}
		add_political_power = -100
	}
}

		}

	}

}

country_event = { # Albanian uprising successful
	id = albania.4
	title = albania.4.t
	desc = albania.4.d

	is_triggered_only = yes

	### this triggered event will only fire if the trigger is fulfilled >>>
	trigger = {
		ALB = {
			exists = yes
			has_war_with = OTT
		}
		OTT = {
			divisions_in_state = {
				state = 44
				size < 1
			}
			divisions_in_state = {
				state = 765
				size < 1
			}
			divisions_in_state = {
				state = 766
				size < 1
			}
		}
	}

	option = { # accept their independence
		name = albania.4.a

		ai_chance = {
			factor = 100
		}
		ALB = {
			white_peace_with_ROOT = yes
		}
		add_political_power = -40
		ALB = {
			country_event = {
				id = albania.6
				days = 1
			}
		}
		clr_country_flag = albanian_revolt
	}

	option = { # never!
		name = albania.4.b

		ai_chance = {
			factor = 0
		}
		add_political_power = -60
	}

}
