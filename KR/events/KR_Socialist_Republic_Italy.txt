	trigger = {
		tag = SRI
		has_socialist_government = yes

			has_completed_focus = SRI_suppress_catholics
			has_war_with = ITA
		}
		ITA = {
			NOT = {
				has_socialist_government = yes
				has_cosmetic_tag = ITA_kingdom
				has_cosmetic_tag = ITA_republic
			}
		}
		NOT = {
			has_idea = SRI_neosanfedisti_1
			has_idea = SRI_neosanfedisti_2
			has_idea = SRI_neosanfedisti_3
			has_country_flag = SRI_neosanfedisti_defeated
			has_capitulated = yes
		}
	}
	
	mean_time_to_happen = {
		days = 30
	}

	option = {
		name = socitaly.20.a
		add_political_power = -100
		add_ideas = SRI_neosanfedisti_1
		set_country_flag = { flag = SRI_neosanfedisti_attacks value = 0 }
	}
}

	id = socitaly.21
	title = socitaly.21.t
	desc = socitaly.21.d
	picture = GFX_report_event_SRI_bombing

	trigger = {
		tag = SRI
		has_socialist_government = yes
		NOT = { has_idea = SRI_neosanfedisti_attack }

	mean_time_to_happen = {
