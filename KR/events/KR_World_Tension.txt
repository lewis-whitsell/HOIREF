add_namespace = worldtension

# Second Weltkrieg
country_event = {
	id = worldtension.1
	title = worldtension.1.t
	desc = worldtension.1.d

	trigger = {
		tag = FRA
		GER = { has_war_with = FRA }
	}

	fire_only_once = yes


	option = {
		name = worldtension.1.a
		add_named_threat = { threat = 25 name = second_weltkrieg_threat }
	}
}
