###########################
# Renaming Events
###########################
#
# Each event id should be 4 digits long
# Add extra digits where necessary - state 12 is 1200 - 1209 in event id's
#
# All events need to be added into on_state_control_changed in KR_on_actions_renaming - use the examples there
#
# Use the first event for a state (for example, '1200') for the original owner
# Then subsequent id's are for other countries that can own & rename that state
#
###########################

add_namespace = rename

### 1 - Corsica
# Restore Italian Name
country_event = {
	id = rename.100
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		OR = {
			tag = ITA
			tag = SRI
		}
		1 = {
			is_controlled_by = ROOT
			has_state_flag = renamed_state
		}
	}

	immediate = {
		hidden_effect = {
			1 = {
				reset_state_name = yes
				clr_state_flag = renamed_state
			}
		}
	}
}
