# Join the Axis

country_event = {
	id = libya.5
	title = libya.5.t
	desc = libya.5.d
	
	fire_only_once = yes
	
	trigger = {
		tag = LYB
		has_war = no
		is_subject = no
		is_in_faction = no
		NOT = {
			has_government = totalist
			has_government = radical_socialist
			has_government = syndicalist
		}
		EGY = {
			exists = yes
			has_war = no
			is_subject = no
			is_in_faction_with = SAU
			is_faction_leader = yes
		}
		SAU = {
			exists = yes
			has_war = no
			is_subject = no
		}
		OTT = { exists = yes }
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	option = {
		name = libya.5.a
		ai_chance = { factor = 90 }

		add_political_power = 50
		EGY = { country_event = { id = libya.6 days = 1 } } # EGY response
	}

	option = {
		name = libya.5.b
		ai_chance = { factor = 10 }

		add_political_power = -50
	}

}
