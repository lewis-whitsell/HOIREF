country_event = {
	id = arabia.4
	title = arabia.4.t
	desc = arabia.4.d

	fire_only_once = yes

	trigger = {
		tag = SAU
		has_global_flag = ara_con2
		date > 1937.3.1
		is_subject = no
		NOT = { controls_state = 754 }
		OTT = {
			NOT = {
				has_war_with = EGY
				has_war_with = SAU
			}
			owns_state = 754
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	option = {					# secretly contact the emir
		name = arabia.4.a

		ai_chance = { factor = 70 }
		add_political_power = 20
		random_list = {
			70 = { country_event = { id = arabia.5 days = 7 } } # emir accepts
			30 = { add_political_power = -10 } # emir rejects
			# TODO - custom tooltip
		}
	}

	option = {					# we cannot trust him
		name = arabia.4.b

		ai_chance = { factor = 30 }
		add_political_power = -20
	}

}

### A New Capital For Hashemite Arabia

country_event = {
	id = arabia.7
	title = arabia.7.t
	desc = arabia.7.d

	fire_only_once = yes

	trigger = {
		tag = SAU
		OR = {
			AND = {
				owns_state = 554 # Damascus
				controls_state = 554
			}
			AND = {
				owns_state = 291 # Baghdad
				controls_state = 291
			}
			AND = {
				owns_state = 454 # Jerusalem
				controls_state = 454
			}
		}
		has_war = no
	}

	mean_time_to_happen = {
		days = 10
	}

	option = {					# mecca
		name = arabia.7.a

		ai_chance = { factor = 5 }
	}

	option = {					# damascus
		name = arabia.7.b
		trigger = {
				owns_state = 554 # damascus
				controls_state = 554
		}

		ai_chance = { factor = 60 }
		set_capital = 554
	}

	option = {					# baghdad
		name = arabia.7.c
		trigger = {
				owns_state = 291 # baghdad
				controls_state = 291
		}

		ai_chance = { factor = 30 }
		set_capital = 291
	}

	option = {					# jerusalem
		name = arabia.7.e
		trigger = {
				owns_state = 454 # jamascus
				controls_state = 454
		}

		ai_chance = { factor = 10 }
		set_capital = 454
	}

}

# YEM - Death of Yahya al Hamid
###############################################################################

### Yahya al Hamid is country leader

country_event = {
	id = arabia.10
	title = arabia.10.t
	desc = arabia.10.d

	fire_only_once = yes

	trigger = {
		tag = YEM
		date > 1948.2.17
		has_country_leader = { ruling_only = yes name = "H.R.M. Yahya al Hamid" } 
	}

	mean_time_to_happen = {
		days = 0
	}

	option = {
		name = arabia.10.a

		if = {
			limit = {
				has_government = paternal_autocrat_subtype
			}

			kill_country_leader = yes
			create_country_leader = {
				name = "H.R.M. Ahmad bin Yahya"
				#desc = ""
				#picture = ""
				expire = "1965.1.1"
				ideology = paternal_autocrat_subtype
				traits = {}
			}

			else = {
				kill_country_leader = yes
				create_country_leader = {
					name = "H.R.M. Ahmad bin Yahya"
					#desc = ""
					#picture = ""
					expire = "1965.1.1"
					ideology = authoritarian_democrat_subtype
					traits = {}
				}
			}
		}
		set_country_flag = Yahya_al_Hamid_dead
		remove_unit_leader = 699
	}

}
