country_event = { # X loses Reichsmark
	id = germit.4
	title = germit.4.t
	desc = germit.4.d
	picture = report_event_generic_read_write

	trigger = {
		has_idea = ME_europamark_idea
		OR = {
			NOT = {
				is_in_faction_with = GER
			}
			NOT = {
				country_exists = GER
			}
			GER = {
				is_subject = yes
			}
		}
	}

	mean_time_to_happen = { days = 2 }

	option = {
		name = germit.4.a
		remove_ideas = ME_europamark_idea
	}
}
