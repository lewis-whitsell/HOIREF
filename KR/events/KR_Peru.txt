	option = {
		name = pru.8.b
		ai_chance = {
			factor = 80
			modifier = {
				factor = 3
				strength_ratio = {
					tag = PRU
					ratio > 0.7
				}
			}
			modifier = {
				factor = 0.1
				any_other_country = {
					is_major = yes
					is_in_faction_with = ROOT
				}
			}
		}
		PRU = {
			country_event = {
				id = pru.18 days = 1
			}
			if = {
				limit = {
					is_in_faction_with = ROOT
					is_faction_leader = yes
				}
				remove_from_faction = ROOT
			}
			if = {
				limit = {
					is_in_faction_with = ROOT
					NOT = { is_faction_leader = yes }
				}
				random_other_country = {
					limit = {
						is_in_faction_with = PRU
						is_faction_leader = yes
					}
					remove_from_faction = PRU
				}
			}
		}
		effect_tooltip = {
			PRU = {
				declare_war_on = {
					target = ROOT
					type = annex_everything
				}
			}
		}
	}
