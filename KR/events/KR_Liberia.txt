# The Freemasons attempt a coup
country_event = {
	id = LBR.10
	title = LBR.10.t 
	desc = LBR.10.d
	
	fire_only_once = yes
	
	trigger = {
		tag = LBR
		has_war = no
		has_country_flag = LBR_societies_legalized
		NOT = { has_government = national_populist }
		NOT = { has_country_flag = LBR_1939_elections_held }
	}
	
	mean_time_to_happen = {
		months = 2
	}
	
	immediate = {
		clr_country_flag = LBR_societies_legalized
	}
	
	option = { #coup succeeds
		name = LBR.10.a
		remove_all_ministers = yes
		add_political_power = -100
		add_ideas = LBR_broken_center
		set_politics = {
			ruling_party = national_populist
			elections_allowed = no
		}
		hidden_effect = {
			add_ideas = LBR_William_Richard_Tolbert_hog_npo
			add_ideas = LBR_Allen_Yancey_for_npo
			add_ideas = LBR_James_Agea_Himie_Jones_eco_npo
			add_ideas = LBR_Cornelius_Bosman_van_Oudkarspel_sec_npo
		}
		add_popularity = { ideology = national_populist popularity = 0.25 }

		# Remove Overlord & Faction
		if = {
			limit = { is_subject = yes }
			overlord = {
				end_puppet = LBR
				country_event = { id = LBR.90 }
			}
		}
		if = {
			limit = { is_in_faction = yes }
			every_country = {
				limit = {
					is_in_faction_with = LBR
					is_faction_leader = yes
				}
				remove_from_faction = LBR
			}
		}
	}
	option = { #coup fails
		name = LBR.10.b
		add_political_power = 50
		
	}
}
