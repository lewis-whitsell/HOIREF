civil_war_militia_production = {
	enable = {
		OR = {
			AND = {
				tag = SOV
				has_war_with = RUS
			}
			AND = {
				tag = TEX
				has_war = yes
				NOT = { has_global_flag = END_ACW }
			}
			AND = {
				tag = CAL
				has_war = yes
				NOT = { has_global_flag = END_ACW }
			}
			AND = {
				tag = CSA
				has_war = yes
				NOT = { has_global_flag = END_ACW }
			}
			AND = {
				tag = USA
				has_war = yes
				NOT = { has_global_flag = END_ACW }
				NOT = { has_global_flag = civil_war_avoided }
			}
			AND = {
				tag = SPR
				OR = {
					has_war_with = CNT
					has_war_with = CAR
				}
			}
			AND = {
				tag = CNT
				OR = {
					has_war_with = SPR
					has_war_with = CAR
				}
			}
			AND = {
				tag = CAR
				OR = {
					has_war_with = CNT
					has_war_with = SPR
				}
			}
		}
	}
	abort = {
		AND = {
			tag = SOV
			NOT = { has_war_with = RUS }
		}
		AND = {
			tag = TEX
			has_global_flag = END_ACW
		}
		AND = {
			tag = CAL
			has_global_flag = END_ACW
		}
		AND = {
			tag = CSA
			has_global_flag = END_ACW
		}
		AND = {
			tag = USA
			OR = {
				has_global_flag = END_ACW
				has_global_flag = civil_war_avoided
			}
		}
		AND = {
			tag = SPR
			NOT = { has_war_with = CNT }
			NOT = { has_war_with = CAR }
		}
		AND = {
			tag = CNT
			NOT = { has_war_with = SPR }
			NOT = { has_war_with = CAR }
		}
		AND = {
			tag = CAR
			NOT = { has_war_with = CNT }
			NOT = { has_war_with = SPR }
		}
	}
	
	ai_strategy = {
		type = role_ratio
		id = infantry
		value = -70
	}

	}

save_admiral_pp_for_later = {
	enable = {
		has_war = no
		threat < 0.3
		has_political_power < 200		
	}
	abort = {
		OR = {
			has_war = yes	
			threat > 0.3
			has_political_power > 200
		}
	}

	ai_strategy = { # dont burn PP on admirals (can still get free ones)
		type = pp_spend_priority
		id = admiral
		value = -100
	}
}
