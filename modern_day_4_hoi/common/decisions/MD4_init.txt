MD4_init = {
	MD4_init_stability = {
		
		allowed = {
			always = yes
		}
		
		available = {
			has_country_flag = MD4_init_stability
			is_ai = yes
		}

		visible = {
			or = {
				and = { has_country_flag = MD4_init_stability is_ai = no }
				and = { has_country_flag = MD4_init_stability is_ai = yes }
				
			}
		}


		Is_good = yes

		selectable_mission = yes
		days_mission_timeout = 1
		days_remove = 30
		

		
		timeout_effect = {
			clr_country_flag = MD4_init_stability
			random_country = {
				limit = { is_ai = yes has_war = no }
				set_country_flag = MD4_init_stability
				country_event = { id = low_stability.1 }
			}
		}


		complete_effect = {
			every_country = {
				country_event = { id = low_stability.1 }
			}
		}
		
		ai_will_do = {
			factor = 100
		}
	}
	MD4_init_debt = {
		
		allowed = {
			always = yes
		}
		
		available = {
			has_country_flag = MD4_init_debt
			is_ai = yes
		}

		visible = {
			or = {
				and = { has_country_flag = MD4_init_debt is_ai = no }
				and = { has_country_flag = MD4_init_debt is_ai = yes }
				
			}
		}


		Is_good = yes

		selectable_mission = yes
		days_mission_timeout = 1
		days_remove = 75
		

		
		timeout_effect = {
			clr_country_flag = MD4_init_debt
			random_country = {
				limit = { is_ai = yes has_war = no }
				set_country_flag = MD4_init_debt
				country_event = { id = debt_count.1 }
			}
		}


		complete_effect = {
			every_country = {
				country_event = { id = debt_count.1 }
			}
		}
		
		ai_will_do = {
			factor = 100
		}
	}
	
}
