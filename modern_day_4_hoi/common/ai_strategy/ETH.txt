#Eritrea-Ethiopia border conflict
Ethiopia_Eritrea_Border_Conflict ={
	enable = {
		original_tag = ETH
		country_exists = ERI
		ERI = { has_government = Nationalist }
		has_opinion_modifier = ERI_Border_Disputes
	}
	abort = {
		OR = {
			NOT = { country_exists = ERI }
			ERI = { NOT = { has_government = Nationalist } }
			NOT = { has_opinion_modifier = ERI_Border_Disputes }
		}
	}
	
	ai_strategy = {
		type = antagonize
		id = "ERI"
		value = 50
	}
	ai_strategy = {
		type = contain
		id = "ERI"
		value = 100
	}
	ai_strategy = {
		type = conquer
		id = "ERI"
		value = 50
	}
}
