Iraq_IRQ_dont_send_volunteers_against_ISIS = {

	enable = {
		original_tag = PER
	}
	
	abort = {				#Allow sending volunteers once ISIS is defeated, or Iraq is at war with someone else
		OR = {
			IRQ = { NOT = { has_war_with = ISI } }
			any_country = {
				NOT = { tag = IRQ }
				NOT = { tag = ISI }
				has_war_with = IRQ 
			}
		}
	}
	
	ai_strategy = {
		type = send_volunteers_desire
		id = "IRQ"
		value = -400
	}
}
