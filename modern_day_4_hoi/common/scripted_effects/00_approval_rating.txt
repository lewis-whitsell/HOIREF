relative_change_pos_2 = {
	if = {
		limit = {
			and = {
				has_country_flag = {
					flag = approval_rating
					value > 79
				}
			}
		}
		modify_country_flag = { flag = approval_rating value = 1 }
		increase = yes approval_rating = yes pos_01 = yes
		approval_rating_update = yes
	}
	if = {
		limit = {
			and = {
				has_country_flag = {
					flag = approval_rating
					value > 59
				}
				has_country_flag = {
					flag = approval_rating
					value < 80
				}
			}
		}
		modify_country_flag = { flag = approval_rating value = 1 }
		increase = yes approval_rating = yes pos_01 = yes
		approval_rating_update = yes
	}
	if = {
		limit = {
			and = {
				has_country_flag = {
					flag = approval_rating
					value > 39
				}
				has_country_flag = {
					flag = approval_rating
					value < 60
				}
			}
		}
		modify_country_flag = { flag = approval_rating value = 2 }
		increase = yes approval_rating = yes pos_02 = yes
		approval_rating_update = yes
	}
	if = {
		limit = {
			and = {
				has_country_flag = {
					flag = approval_rating
					value > 19
				}
				has_country_flag = {
					flag = approval_rating
					value < 40
				}
			}
		}
		modify_country_flag = { flag = approval_rating value = 3 }
		increase = yes approval_rating = yes pos_03 = yes
		approval_rating_update = yes
	}
	if = {
		limit = {
				has_country_flag = {
					flag = approval_rating
					value < 20
				}
		}
		modify_country_flag = { flag = approval_rating value = 4 }
		increase = yes approval_rating = yes pos_04 = yes
		approval_rating_update = yes
	}
}
