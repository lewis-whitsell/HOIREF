scripted_gui = {

	money = {
		context_type = player_context
		
		window_name = "money_container"
		
		visible = {
            has_money = yes
		}
		
		effects = {
			money_bg_click = {
				set_temp_variable = { show = 1 }
				subtract_from_temp_variable = { show = show_money_stuff }
				set_variable = { show_money_stuff = show }
			}	
            
			money_bg_right_click = {
				add_to_variable = { money = 100 }
                set_variable = { money_max = 500 }
			}
		}
	}
    
}
