add_namespace = city_captured

#Fall of Allepo
news_event = {
	id = city_captured.0
	title = city_captured.0.t
	desc = city_captured.0.d
	picture = GFX_news_mideast_citycapture
	
	major = yes
	
	trigger = {
		not = { 566 = { is_controlled_by = SYR } }
		not = { has_global_flag = captured_allepo_global }
		SYR = { has_war = yes}
		566 = { is_controlled_by = ROOT }
	}
	
	immediate = {
		set_global_flag = captured_allepo_global
	}

	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = city_captured.0.a
		Allepo_captured = yes #Gives 1% boost to org & morale for 60 days to either NUS/FSA/ISI/ROJ
		effect_tooltip = {
			from = {
				add_timed_idea = { idea = Strategic_city_captured1 days = 60 }
			}
		}
	}
	
	
}
