add_namespace = Somalia
add_namespace = SomaliaNews

### Somalia civil war cleanup events ###

#Give the winner SOM cosmetic tag
country_event = {
	id = Somalia.0
	
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				tag = SOM
				NOT = {
					country_exists = SHB 
				}
			}
			AND = {
				tag = SHB
				NOT = {
					country_exists = SOM 
				}
			}
		}
		NOT = { has_global_flag = Somalia_Civil_War_Over }
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		#Civil war is over
		set_global_flag = Somalia_Civil_War_Over
		#Clear core status
		240 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		239 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		238 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		237 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		#Make the winner Somalia
		if = {
			limit = { tag = SHB	}
			set_cosmetic_tag = SOM
		}
		add_state_core = 240
		add_state_core = 239
		add_state_core = 238
		add_state_core = 237
		set_capital = 240
		#Mark who won
		if = {
			limit = { tag = SHB }
			set_country_flag = Somalia_Civil_War_Islamist_Victory
			news_event = { id = SomaliaNews.0 hours = 6 }
		}
		if = {
			limit = { tag = SOM }
			set_country_flag = Somalia_Civil_War_Government_Victory
			news_event = { id = SomaliaNews.1 hours = 6 }
		}
	}
	
	option = {
		name = Somalia.0.a
	}
}
