#Kurds are Supressed
country_event = {
    id = kurd_rally.3
    title = kurd_rally.3.t
    desc = kurd_rally.3.d
	picture = GFX_kurd_riot
	
	fire_only_once = yes
	is_triggered_only = yes

	#Good - trigger riot and PKK attack
    option = {
    name = kurd_rally.3.a
	custom_effect_tooltip = kurd_rally_tt.4
	hidden_effect = {
	set_country_flag = kurd_riot1
	    if = {
    		limit = {
			    NOT = {
    	            TUR = {
    	        	    has_country_flag = tur_security_increase
    	                }
    	            }
				}
				TUR = {
    	    	    set_country_flag = pkk_attack_win
    	            }
    	    }
	    if = {
    		limit = {
    	        TUR = {
    	    	    has_country_flag = tur_security_increase
    	            }
    	        }
				TUR = {
    	    	    set_country_flag = pkk_attack_lose
    	            }
    	    }
        }
    }
}

#PKK Ambush - Successful
country_event = {
    id = pkk_attack.1
    title = pkk_attack.1.t
    desc = pkk_attack.1.d
	picture = GFX_pkk_ambush
	
	fire_only_once = yes
	
	trigger = {
	tag = TUR
	OR = {
	    AND = {
	        TUR = {
	            controls_state = 162
	            has_idea = multi_ethnic_state_idea
	            has_country_flag = pkk_attack_win
	        	has_country_flag = kurd_activism
	               }
	            PKK = {
	            exists = no
	            }
	        }
	    AND = {
		    TUR = {
	            controls_state = 162
	            has_idea = multi_ethnic_state_idea
	        	has_country_flag = kurd_activism
				has_stability < 0.5
	               }
	            PKK = {
	            exists = no
	            }
		    }
	    }
	}

		mean_time_to_happen = {
		days = 80
	}
	
	immediate = {
	set_country_flag = pkk_win
	}

	#Investigate
    option = {
    name = pkk_attack.1.a
    }
}
