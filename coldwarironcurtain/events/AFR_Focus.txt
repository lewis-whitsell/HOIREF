# independence Refused
country_event = {
	id = africa.2
	title = africa.2.t
	desc = africa.2.d

	is_triggered_only = yes

	option = {
		name = "This means war!"
		set_autonomy = { target = ROOT autonomy_state = autonomy_free }	
		declare_war_on = {
			target = FROM
			type = civil_war
		}
		if = {
			limit = {
				has_idea = nato_member
			}
			remove_ideas = nato_member
		}
		add_political_power = 250
		add_war_support = 0.3
		ai_chance = {
			factor = 50
			modifier = {
				add = 10
				check_variable = {
					ROOT.Anger > 9
				}
			}
			modifier = {
				add = 20
				check_variable = {
					ROOT.Anger > 19
				}
			}
			modifier = {
				add = 30
				check_variable = {
					ROOT.Anger > 29
				}
			}
		}
	}

	option = {
		name = "It's not over yet"
		custom_effect_tooltip = "We'll ask them again in about a year. Maybe their decision will change..."
		hidden_effect = {
			OVERLORD = { country_event = { id = africa.1 days = 360 random = 1 } }
			add_to_variable = { var = ROOT.Anger value = 10 }
		}
		ai_chance = {
			factor = 50
			modifier = {
				add = -10
				check_variable = {
					ROOT.Anger > 9
				}
			}
			modifier = {
				add = -20
				check_variable = {
					ROOT.Anger > 19
				}
			}
			modifier = {
				add = -30
				check_variable = {
					ROOT.Anger > 29
				}
			}
		}
	}
}
